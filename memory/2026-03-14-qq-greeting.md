```bash
# 1. 创建 SSL 目录
mkdir -p /etc/nginx/ssl

# 2. 写入证书（把你的真实证书内容替换下面）
cat > /etc/nginx/ssl/qqbot.finloan.cn_bundle.crt << 'CERTEOF'
-----BEGIN CERTIFICATE-----

...（你的证书内容）...
-----END CERTIFICATE-----
CERTEOF

# 3. 写入私钥（把你的真实私钥内容替换下面）
cat > /etc/nginx/ssl/qqbot.finloan.cn.key << 'KEYEOF'
-----BEGIN RSA PRIVATE KEY-----
MIIEowIBAAKCAQEA0tZyKUHDW3DCv6wpYGYBP0H1TGiZzmaB+9hEagOatFMaGC96
...（你的私钥内容）...
-----END RSA PRIVATE KEY-----
KEYEOF

# 4. 写入 Nginx 配置（使用 conf.d 目录更通用）
cat > /etc/nginx/conf.d/qqbot.conf << 'NGINXEOF'
server {
    listen 443 ssl;
    server_name qqbot.finloan.cn;
    
    ssl_certificate /etc/nginx/ssl/qqbot.finloan.cn_bundle.crt;
    ssl_certificate_key /etc/nginx/ssl/qqbot.finloan.cn.key;
    
    location / {
        proxy_pass http://localhost:8080;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }
}
NGINXEOF

# 5. 测试配置
nginx -t

# 6. 如果测试通过，重启 Nginx
systemctl restart nginx
```
